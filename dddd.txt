    struct DbInfo
    {
        public string DbName { get; }
        public string TblName { get; }

        public DbInfo(string dbName, string tblName)
        {
            DbName = dbName;
            TblName = tblName;
        }

        public static bool operator ==(DbInfo a, DbInfo b)
        {
            return a.DbName == b.DbName && a.TblName == b.TblName;
        }

        public static bool operator !=(DbInfo a, DbInfo b)
        {
            return !(a == b);
        }
    }
